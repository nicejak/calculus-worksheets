    \begin{enumerate}
        \item Suppose $f(x)$ is differentiable everywhere and invertible with domain $[-17,\infty)$, range $[-2,4]$, and evaluations and derivatives defined by the following:
        \begin{center}
            \begin{tabular}{ c  c }
                $f(1)=4$            &   $f'(1)=-1$\\[.5em]
                $f(2)=2$            &   $f'(2)=-\frac{1}{2}$\\[.5em]
                $f(3)=1$            &   $f'(3)=-\frac{1}{4}$\\[.5em]
                $f(4)=\frac{1}{2}$  &   $f'(4)=-\frac{1}{5}$\\[.5em]
            \end{tabular}    
        \end{center}
        \begin{itemize}
            \item What is the domain of $f^{-1}(x)$?\vfill
            \item What is the range of $f^{-1}(x)$?\vfill
            \item What is $(f^{-1})'(1)$?\vfill
        \end{itemize}
        \newpage
        \item What is $$\arcsin\left(\sin\left(\frac{11\pi}{3}\right)\right)?$$\vfill
        \item What is $$\csc\left(\arctan\left(\frac{4}{3}\right)\right)?$$\vfill
        \item Calculate the limit $$\lim_{x\to0}x^{-1}\sin(x^{2}).$$\vfill
    \end{enumerate}
